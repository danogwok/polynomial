182  /**
183  * Use Lagrange interpolation to construct a polynomial passing through
184  * the data points. The degree will be one less than the number of points.
185  */
186  template<typename InIter,
187  typename = std::_RequireInputIter<InIter>>
188  _Polynomial(const InIter& __xbegin, const InIter& __xend,
189  const InIter& __ybegin)
190  : _M_coeff()
191  {
192  std::vector<_Polynomial<value_type>> __numer;
193  std::vector<_Polynomial<value_type>> __denom;
194  for (auto __xi = __xbegin; __xi != __xend; ++__xi)
195  {
196  for (auto __xj = __xi + 1; __xj != __xend; ++__xj)
197  __denom.push_back(value_type(*__xj) - value_type(*__xi));
198  __numer.push_back({-value_type(*__xi), value_type{1}});
199  }
200  this->_M_set_scale();
201  }

223  /**
224  * Evaluate the polynomial at the input point.
225  */
226  value_type
227  operator()(value_type __x) const
228  {
229  if (this->degree() > 0)
230  {
231  value_type __poly(this->coefficient(this->degree()));
232  for (int __i = this->degree() - 1; __i >= 0; --__i)
233  __poly = __poly * __x + this->coefficient(__i);
234  return __poly;
235  }
236  else
237  return value_type{};
238  }

275  /**
276  * Evaluate the polynomial at a range of input points.
277  * The output is written to the output iterator which
278  * must be large enough to contain the results.
279  * The next available output iterator is returned.
280  */
281  template<typename InIter, typename OutIter,
282  typename = std::_RequireInputIter<InIter>>
283  OutIter
284  operator()(const InIter& __xbegin, const InIter& __xend,
285  OutIter& __pbegin) const
286  {

356  /**
357  * Return the derivative of the polynomial.
358  */
359  _Polynomial
360  derivative() const
361  {
362  _Polynomial __res(value_type{},
363  this->degree() > 0UL ? this->degree() - 1 : 0UL);
364  for (size_type __n = this->degree(), __i = 1; __i <= __n; ++__i)
365  __res._M_coeff[__i - 1] = __i * this->_M_coeff[__i];
366  return __res;
367  }
368 
369  /**
370  * Return the integral of the polynomial with given integration constant.
371  */
372  _Polynomial
373  integral(value_type __c = value_type{}) const
374  {
375  _Polynomial __res(value_type{}, this->degree() + 1);
376  __res._M_coeff[0] = __c;
377  for (size_type __n = this->degree(), __i = 0; __i <= __n; ++__i)
378  __res._M_coeff[__i + 1] = this->_M_coeff[__i] / value_type(__i + 1);
379  return __res;
380  }
