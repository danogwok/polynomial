189  /**
190  * The following polynomial evaluations are done using
191  * a modified of Horner's rule which exploits the fact that
192  * the polynomial coefficients are all real.
193  * The algorithm is discussed in detail in:
194  * Knuth, D. E., The Art of Computer Programming: Seminumerical
195  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
196  *
197  * If n is the degree of the polynomial, n - 3 multiplies are
198  * saved and 4 * n - 6 additions are saved.
199  */
200  template<typename _Tp2>
201  constexpr auto
202  operator()(std::complex<_Tp2> __z) const
203  -> decltype(value_type{} * std::complex<_Tp2>{})
204  {
205  const auto __r = _Tp{2} * std::real(__z);
206  const auto __s = std::norm(__z);
207  auto __aa = this->coefficient(this->degree());
208  auto __bb = this->coefficient(this->degree() - 1);
209  for (int __j = 1; __j <= this->degree(); ++__j)
210  {
211  auto __cc = __s * __aa;
212  __aa = __bb + __r * __aa;
213  __bb = this->coefficient(this->degree() - __j) - __cc;
214  }
215  return __aa * __z + __bb;
216  };

235  // Could/should this be done by output iterator range?
236  template<size_type N>
237  constexpr void
238  eval(value_type __x, std::array<value_type, N>& __arr)
239  {
240  if (__arr.size() > 0)
241  {
242  __arr.fill(value_type{});
243  const size_type __sz = _M_coeff.size();
244  __arr[0] = this->coefficient(__sz - 1);
245  for (int __i = __sz - 2; __i >= 0; --__i)
246  {
247  int __nn = std::min(__arr.size() - 1, __sz - 1 - __i);
248  for (int __j = __nn; __j >= 1; --__j)
249  __arr[__j] = __arr[__j] * __x + __arr[__j - 1];
250  __arr[0] = __arr[0] * __x + this->coefficient(__i);
251  }
252  // Now put in the factorials.
253  value_type __fact = value_type(1);
254  for (size_t __i = 2; __i < __arr.size(); ++__i)
255  {
256  __fact *= value_type(__i);
257  __arr[__i] *= __fact;
258  }
259  }
260  }

262  /**
263  * Evaluate the polynomial and its derivatives at the point x.
264  * The values are placed in the output range starting with the
265  * polynomial value and continuing through higher derivatives.
266  */
267  template<typename OutIter>
268  constexpr void
269  eval(value_type __x, OutIter __b, OutIter __e)
270  {
271  if(__b != __e)
272  {
273  std::fill(__b, __e, value_type{});
274  constexpr size_type __sz = _M_coeff.size();
275  *__b = _M_coeff[__sz - 1];
276  for (int __i = __sz - 2; __i >= 0; --__i)
277  {
278  for (auto __it = std::reverse_iterator<OutIter>(__e);
279  __it != std::reverse_iterator<OutIter>(__b) - 1; ++__it)
280  *__it = *__it * __x + *(__it + 1);
281  *__b = *__b * __x + _M_coeff[__i];
282  }
283  // Now put in the factorials.
284  int __i = 0;
285  value_type __fact = value_type(++__i);
286  for (auto __it = __b + 1; __it != __e; ++__it)
287  {
288  __fact *= value_type(__i);
289  *__it *= __fact;
290  ++__i;
291  }
292  }
293  }

331  /**
332  * Evaluate the even part of the polynomial using a modification
333  * of Horner's rule which exploits the fact that the polynomial
334  * coefficients are all real.
335  *
336  * The algorithm is discussed in detail in:
337  * Knuth, D. E., The Art of Computer Programming: Seminumerical
338  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
339  *
340  * If n is the degree of the polynomial,
341  * n - 3 multiplies and 4 * n - 6 additions are saved.
342  */
343  template<typename _Tp2>
344  auto
345  eval_even(std::complex<_Tp2> __z) const
346  -> decltype(value_type{} * std::complex<_Tp2>{})
347  {
348  if (this->degree() > 0)
349  {
350  const auto __zz = __z * __z;
351  const auto __r = _Tp{2} * std::real(__zz);
352  const auto __s = std::norm(__zz);
353  auto __odd = this->degree() % 2;
354  size_type __n = this->degree() - __odd;
355  auto __aa = this->coefficient(__n);
356  auto __bb = this->coefficient(__n - 2);
357  for (size_type __j = 4; __j <= __n; __j += 2)
358  __bb = this->coefficient(__n - __j)
359  - __s * std::exchange(__aa, __bb + __r * __aa);
360  return __aa * __zz + __bb;
361  }
362  else
363  return decltype(value_type{} * std::complex<_Tp2>{}){};
364  };

366  /**
367  * Evaluate the odd part of the polynomial using a modification
368  * of Horner's rule which exploits the fact that the polynomial
369  * coefficients are all real.
370  *
371  * The algorithm is discussed in detail in:
372  * Knuth, D. E., The Art of Computer Programming: Seminumerical
373  * Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
374  *
375  * If n is the degree of the polynomial,
376  * n - 3 multiplies and 4 * n - 6 additions are saved.
377  */
378  template<typename _Tp2>
379  auto
380  eval_odd(std::complex<_Tp2> __z) const
381  -> decltype(value_type{} * std::complex<_Tp2>{})
382  {
383  if (this->degree() > 0)
384  {
385  const auto __zz = __z * __z;
386  const auto __r = _Tp{2} * std::real(__zz);
387  const auto __s = std::norm(__zz);
388  auto __even = (this->degree() % 2 == 0 ? 1 : 0);
389  size_type __n = this->degree() - __even;
390  auto __aa = this->coefficient(__n);
391  auto __bb = this->coefficient(__n - 2);
392  for (size_type __j = 4; __j <= __n; __j += 2)
